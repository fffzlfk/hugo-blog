### 问题描述
请设计递归算法求以元素为x根节点的子树深度

### 代码实现
```c
#include <stdio.h>
#include <stdlib.h>

#define OK 1
#define ERROR 0
typedef char ElemType;
typedef int Status;
typedef struct Node {
	ElemType data;
	Node *left;
	Node *right;
} * BTree;

Status createBTree(BTree &T) {
	ElemType c;
	scanf("%c", &c);
	getchar();
	if (c == ' ') T = NULL;
	else {
		if (!(T = (Node*)malloc(sizeof(Node))))
			return ERROR;
		T->data = c;
		printf("请输入%c的左孩子\n", c);
		createBTree(T->left);
		printf("请输入%c的右孩子\n", c);
		createBTree(T->right);
	}
}

int getDepth(BTree T) {
	if (T == NULL)
		return 0;
	else {
		return getDepth(T->left) > getDepth(T->right) ? getDepth(T->left) + 1 : getDepth(T->right) + 1;
	}
}

int getChildDepth(BTree T, ElemType e) {
	if (T) {
		if (T->data == e) {
			return getDepth(T);
		}
		else {
			int m = getChildDepth(T->left, e);
			int n = getChildDepth(T->right, e);
			return m >= n ? m : n;
		}
	} else
		return 0;
}

void preOrder(BTree T) {
	if (T != NULL)
		printf("%c\n", T->data);
	else
		return;
	preOrder(T->left);
	preOrder(T->right);
}

int main(int argc, char const *argv[]) {
	BTree T;
	createBTree(T);
	printf("%d\n", getChildDepth(T, '3'));
	preOrder(T);
	return 0;
}
```
