### 问题描述
* 分别使用邻接表和邻接矩阵存储无向图图，并设计深度优先搜索遍历算法和广度优先搜索遍历算法。
* 用1,2,3，...，表示结点
* 并分别求出深度优先生成树和广度优先生成树的边集。

### 代码实现

###### 邻接表
```cpp
#include <stdio.h>
#include <stdlib.h>

#define MAX_VERTEX_NUM 31
typedef int InfoType;
typedef int VertexType;
bool visited[MAX_VERTEX_NUM];
bool dfs_arc[MAX_VERTEX_NUM][MAX_VERTEX_NUM];
bool bfs_arc[MAX_VERTEX_NUM][MAX_VERTEX_NUM];
typedef struct ArcNode {
	int adjvex;
	ArcNode *nextarc;
	InfoType *info;
} ArcNode;

typedef struct VNode {
	VertexType data;
	ArcNode *firstarc;
} VNode, AdjList[MAX_VERTEX_NUM];

typedef struct {
	AdjList vertices;
	int vexnum, arcnum;
	int kind;
} ALGraph;
//创建无向图
void Create(ALGraph &G) {
	puts("输入节点数和边数");
	scanf("%d%d", &G.vexnum, &G.arcnum);
	for (int i = 1; i <= G.vexnum; i++) {
		G.vertices[i].data = i;
		G.vertices[i].firstarc = NULL;
	}
	puts("依次输入弧");
	for (int i = 1; i <= G.arcnum; i++) {
		int v1, v2;
		scanf("%d%d", &v1, &v2);
		ArcNode *p = (ArcNode*)malloc(sizeof(ArcNode));
		p->adjvex = v2;
		p->nextarc = G.vertices[v1].firstarc;
		G.vertices[v1].firstarc = p;
		p = (ArcNode*)malloc(sizeof(ArcNode));
		p->adjvex = v1;
		p->nextarc = G.vertices[v2].firstarc;
		G.vertices[v2].firstarc = p;
	}
}

void DFS(ALGraph *G, int v) {
	ArcNode *p;
	visited[v] = 1;
	printf("当前DFS节点为%d\n", G->vertices[v].data);
	p = G->vertices[v].firstarc;
	while (p) {
		if (!visited[p->adjvex]) {
			dfs_arc[v][p->adjvex] = 1;
			dfs_arc[p->adjvex][v] = 1;
			DFS(G, p->adjvex);
		}
		p = p->nextarc;
	}
}

void DFS_traverse(ALGraph *G) {
	for (int k = 1; k <= G->vexnum; k++)
		visited[k] = 0;
	puts("DFS");
	int v;
	puts("请输入DFS起点");
	scanf("%d", &v);
	DFS(G, v);
	puts("DFS生成树边集:");
	for (int i = 1; i <= G->vexnum; i++) {
		for (int j = i; j <= G->vexnum; j++) {
			if (dfs_arc[i][j])
				printf("%d %d\n", i, j);
		}
	}
}

void BFS_traverse(ALGraph *G) {
	puts("BFS");
	int Q[MAX_VERTEX_NUM];
	int front = 0;
	int rear = 0;
	for (int k = 1; k <= G->vexnum; k++)
		visited[k] = 0;
	int v;
	puts("请输入BFS起点");
	scanf("%d", &v);
	printf("当前BFS节点为%d\n", v);
	visited[v] = 1;
	Q[++rear] = v;
	while (front != rear) {
		int w = Q[++front];
		ArcNode *p = G->vertices[w].firstarc;
		while (p) {
			if (!visited[p->adjvex]) {
				printf("当前BFS节点为%d\n", p->adjvex);
				visited[p->adjvex] = 1;
				Q[++rear] = p->adjvex;
				bfs_arc[w][p->adjvex] = 1;
				bfs_arc[p->adjvex][w] = 1;
			}
			p = p->nextarc;
		}
	}

	puts("BFS生成树边集：");
	for (int i = 1; i <= G->vexnum; i++) {
		for (int j = i; j <= G->vexnum; j++) {
			if (bfs_arc[i][j])
				printf("%d %d\n", i, j);
		}
	}
}

int main(int argc, char const *argv[]) {
	ALGraph G;
	Create(G);
	DFS_traverse(&G);
	BFS_traverse(&G);
	return 0;
}
```

###### 邻接矩阵
```cpp
#include <iostream>
#include <queue>
using namespace std;
const int MAX = 30;
int arc[MAX][MAX];
bool visit[MAX];
bool dfs_arc[MAX][MAX];
bool bfs_arc[MAX][MAX];
int n;
int m;
void dfs(int v) {
	if (v <= n)
		printf("%d\n", v);
	visit[v] = 1;
	for (int i = 1; i <= n; i++) {
		if (arc[v][i] == 1 && visit[i] == 0) {
			dfs_arc[v][i] = 1;
			dfs_arc[i][v] = 1;
			dfs(i);
		}
	}
}

void dfs_traverse() {
	for (int i = 1; i <= n; i++)
		visit[i] = 0;
	int v;
	puts("请输入DFS起点");
	scanf("%d", &v);
	dfs(v);
	puts("DFS生成树边集为");
	for (int i = 1; i <= n; i++) {
		for (int j = i; j <= n; j++) {
			if (dfs_arc[i][j])
				printf("%d %d\n", i, j);
		}
	}
}

void bfs_traverse() {
	for (int i = 1; i <= n; i++)
		visit[i] = 0;
	queue<int> Q;
	int v;
	puts("请输入BFS起点");
	cin >> v;
	Q.push(v);
	printf("%d\n", v);
	visit[v] = 1;
	while (!Q.empty()) {
		v = Q.front();
		Q.pop();
		for (int i = 1; i <= n; i++) {
			if (arc[v][i] == 1 && !visit[i]) {
				printf("%d\n", i);
				bfs_arc[v][i] = 1;
				bfs_arc[i][v] = 1;
                Q.push(i);
			}
		}
	}
	puts("BFS生成树边集为");
	for (int i = 1; i <= n; i++) {
		for (int j = i; j <= n; j++) {
			if (bfs_arc[i][j])
				printf("%d %d\n", i, j);
		}
	}
}

int main(int argc, char const *argv[]) {
	puts("输入节点数和边数");
	cin >> n >> m;
	int v1, v2;
	puts("依次输入各边");
	for (int i = 0; i < m; i++) {
		cin >> v1 >> v2;
		arc[v1][v2] = 1;
		arc[v2][v1] = 1;
	}
	dfs_traverse();
	bfs_traverse();
	return 0;
}
```
