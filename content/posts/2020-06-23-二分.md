### P1024 一元三次方程求解【浮点数二分模版】
<h4><a href="https://www.luogu.com.cn/problem/P1024">题目链接</a></h4>

#### AC代码
```cpp
#include <bits/stdc++.h>
using namespace std;
double A, B, C, D;
double check(double x) { return A * pow(x, 3) + B * pow(x, 2) + C * x + D; }

void bs() {
    double l, r, mid;
    int cnt = 0;
    for (double i = -100.0; i <= 100.0; i++) {
        l = i, r = i + 1;
        double chl = check(l);
        double chr = check(r);
        if (chl == 0) {
            printf("%.2lf ", l);
            cnt++;
        }
        if (chl * chr < 0) {
            while (l + 0.001 <= r) {
                mid = (l + r) / 2;
                if (check(mid) * check(r) <= 0) {
                    l = mid;
                } else {
                    r = mid;
                }
            }
            printf("%.2lf ", l);
            cnt++;
        }
        if (cnt == 3) break;
    }
}

int main() {
    cin >> A >> B >> C >> D;
    bs();
    system("pause");
    return 0;
}
```
### P2678 跳石头【二分模版】
<h4><a href="https://www.luogu.com.cn/problem/P2678">题目链接</a></h4>

#### AC代码
```cpp
#include <bits/stdc++.h>
using namespace std;
const int maxn = 50005;
int d[maxn];
int n, l, m;

bool judge(int step) {
    int cnt = 0;
    int cur = 0;
    int next = 0;
    while (next < n + 1) {
        next++;
        if (d[next] - d[cur] < step) {
            cnt++;
        } else {
            cur = next;
        }
    }
    return cnt <= m;
}

int bs(int _l, int r) {
    int mid, ans;
    while (_l <= r) {
        mid = (_l + r) / 2;
        if (judge(mid)) {
            ans = mid;
            _l = mid + 1;
        } else {
            r = mid - 1;
        }
    }
    return ans;
}

int main() {
    cin >> l >> n >> m;
    int t;
    for (int i = 1; i <= n; i++) {
        cin >> d[i];
    }
    d[n + 1] = l;
    cout << bs(1, l) << '\n';
    system("pause");
    return 0;
}
```
