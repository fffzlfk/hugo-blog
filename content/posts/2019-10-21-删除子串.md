### 问题描述
删除s中所有与t相同的子串
>如 s = "abcdabcdabcd", t = "cd", 则操作后s = "ababab"

### 代码实现
```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int* GetNext(string p) {
  int pLen = p.length();
  int *next = new int[pLen];
  next[0] = -1;
  int j = 0;
  int k = -1;
  while (j < pLen - 1) {
    if (k == -1 || p[j] == p[k]) {
      j++;
      k++;
      next[j] = k;
    } else {
      j = next[j];
    }
  }
  return next;
}

void solve(string &s, string p) {
  int* next =GetNext(p);
  int sLen = s.length();
  int pLen = p.length();
  int i = 0;
  int j = 0;
  while (i < sLen && j < pLen) {
    if (j == -1 || s[i] == p[j]) {
      i++;
      j++;
    } else {
      j = next[j];
    }
  }
  if (j == pLen) {
    s.erase(i-j, j);
    solve(s, p);
  } else {
    return;
  }
}

int main(int argc, char const *argv[]) {
  string s, p;
  cin >> s >> p;
  solve(s, p);
  cout << s << '\n';
  return 0;
}
```
