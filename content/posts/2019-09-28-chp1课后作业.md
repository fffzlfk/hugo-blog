```cpp
typedef int Status;
typedef int ElemType;
typedef struct {
	ElemType *elem;
	int length;
	int listsize;
} SqList;

Status InitList_Sq(SqList &L) {
	//构造一个空线性表
	L.elem = (ElemType*)malloc(LIST_INIT_SIZE * sizeof(ElemType));
	if (!L.elem) exit(OVERFLOW);
	L.length = 0;
	L.listsize = LIST_INIT_SIZE;
	return OK;
}

Status ListInsert_Sq(SqList &L, int i, ElemType e) {
	//在顺序线性表L中的第i个位置之前插入新的元素e
	if (i < 1 || i > L.length + 1) return ERROR;
	if (L.length > L.listsize) {
		ElemType *newbase = (ElemType*) realloc(L.elem, (L.listsize + LISTINCREMENT) * sizeof (ElemType));
		if (!newbase) exit(OVERFLOW);
		L.elem = newbase;
		L.listsize += LISTINCREMENT;
	}
	ElemType *q = &(L.elem[i - 1]);
	for (ElemType *p = &(L.elem[L.length - 1]); p >= q; --p)
		*(p + 1) = *p;
	*q = e;
	L.length++;
	return OK;
}

Status InsertOrderList(SqList &L, ElemType x) {
	if (L.length == L.listsize) {
		return OVERFLOW;
	}
	else {
		int i = L.length - 1;
		for (; i >= 0 && x < L.elem[i]; i--);
		for (int j = L.length - 1; j >= i + 1; j--) {
			L.elem[j + 1] = L.elem[j];
		}
		L.elem[i + 1] = x;
		L.length++;
		return OK;
	}
}

void ShowSqList(SqList L) {
	putchar('[');
	for (int i = 0; i < L.length - 1; i++) {
		printf("%d,", *(L.elem + i));
	}
	printf("%d]\n", *(L.elem + L.length - 1));
}

int main() {
	SqList l;
	InitList_Sq(l);
	ListInsert_Sq(l, 1, 2);
	ListInsert_Sq(l, 2, 3);
	ListInsert_Sq(l, 3, 4);
	ShowSqList(l);
	InsertOrderList(l, 3);
	ShowSqList(l);
	return 0;
}
```

```cpp
#define TRUE 1 
#define FALSE 0

#define ERROR 0

typedef struct node {
    ElemType data;
    struct node* next;

    LinkList head = (LinkList)malloc(sizeof(LinkList));
    LinkList end;
    if (!head) exit(OVERFLOW);
    end = head;

        end = p;
    }
    end->next = NULL;
    L = head;
    return OK;
}

int Length(LinkList L) {
    int length = 0;
    while (L->next != NULL) {
        L = L->next;
        length++;
    }
    return length;
}
void Show(LinkList L) {
    while (L->next != NULL) {
        L = L->next;
        printf("%d\n", L->data);
    }
}

int main() {
    LinkList L;
    int n = 5;
    CreateLinkList(L, 5);
    Show(L);
    printf("%d", Length(L));
    return 0;
```

```cpp
Status Delete(LinkList &L, ElemType mink, ElemType maxk) {
    node *p, *p1 = L;
    while (p1->next != NULL) {
        if (p1->next->data > mink && p1->next->data < maxk) {
            p = p1->next;
            p1->next = p1->next->next;
            free(p);
        }
        else p1 = p1->next;
    }
    return OK;
}

void Show(LinkList L) {
    while (L->next != NULL) {
        L = L->next;
        printf("%d\n", L->data);
    }

{
    LinkList L;
    int n;
    scanf("%d", &n);
    CreateLinkList(L, n);
    Show(L);
    Delete(L, 2, 5);
```

### 倒置链表

```cpp
#include <stdio.h>
#include <stdlib.h>

#define TRUE 1 
#define FALSE 0 
#define OK 1 
#define ERROR 0 
#define INFEASIBLE -1 
#define OVERFLOW -2

typedef int Status;
typedef int ElemType;
typedef struct node {
    ElemType data;
    struct node* next;
} *LinkList;

Status CreateLinkList(LinkList &L, int n) {
    LinkList head = (LinkList)malloc(sizeof(LinkList));
    LinkList end;
    if (!head) exit(OVERFLOW);
    end = head;
    for (int i = 0; i < n; i++) {
        node* p = (node*)malloc(sizeof(node));
        scanf("%d", &p->data);
        end->next = p;
        end = p;
    }
    end->next = NULL;
    L = head;
    return OK;
}

Status ReverseLinkList(LinkList &L) { 
    if (L == NULL || L->next == NULL)
        return OK;
    node *prev = L;
    node *cur = L->next;
    node *tmp = L->next->next;
    while (tmp) {
        tmp = cur->next;
        cur->next = prev;
        prev = cur;
        cur = tmp;
    }
    L->next->next = NULL;
    L->next = prev;
    return OK;
}

void Show(LinkList L) {
    while (L->next != NULL) {
        L = L->next;
        printf("%d\n", L->data);
    }
}

int main(int argc, char const *argv[])
{
    LinkList L;
    int n;
    scanf("%d", &n);
    CreateLinkList(L, n);
    Show(L);
    ReverseLinkList(L);
    Show(L);
    return 0;
}
```
