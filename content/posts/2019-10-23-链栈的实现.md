### 简介
链栈一般用一个头指针结构top来表示，它指向栈顶元素，对于链表来说，top结构相当于它的头指针head。

### 代码实现
```c
#include <stdio.h>
#include <stdlib.h>

struct Node {
	int data;
	Node *next;
};

typedef struct {
	Node *top;
	int count;
} * Stack;

void InitStack(Stack &s) {
	s->top = (Node*)malloc(sizeof(Node));
	if (!s->top) {
		puts("ERROR");
		return;
	}
	s->top = NULL;
	s->count = 0;
}

void Push(Stack &s, int e) {
	Node *node = (Node*)malloc(sizeof(Node));
	node->data = e;
	node->next = s->top;
	s->top = node;
	s->count++;
}

int Pop(Stack &s) {
	if (s->count == 0) {
		puts("Empty");
		exit(0);
	}
	Node *t = s->top;
	s->top = s->top->next;
	int e = t->data;
	free(t);
	s->count--;
	return e;
}

void ClearStack(Stack &s) {
	if (s->count == 0) return;
	Node *p, *q;
	p = s->top;
	while (p) {
		q = p;
		p = p->next;
		free(q);
	}
	s->count = 0;
}

int main(int argc, char const *argv[]) {
	Stack s;
	InitStack(s);
	for (int i = 0; i < 10; i++)
		Push(s, i);
	// ClearStack(s);
	printf("%d\n", Pop(s));
	Push(s, 100);
	for (int i = 0; i < 11; i++) {
		printf("%d\n", Pop(s));
	}
	return 0;
}
```
