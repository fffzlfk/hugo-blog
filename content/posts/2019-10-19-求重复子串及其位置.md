### 问题描述
给出一个线性排列的串s，求出s中出现的第一个最长重复子串及其位置；并分析时间复杂度。
>例如 s = "abcdbcd"，它的第一个出现的最长重复子串及其位置分别为："bcd，1"。

### 代码实现
```cpp
#include <iostream>
using namespace std;

int* GetNext(string p) {
	int pLen = p.length();
	int* next = new int[pLen];
	next[0] = -1;
	int j = 0;
	int k = -1;
	while (j < pLen - 1) {
		if (k == -1 || p[k] == p[j]) {
			k++;
			j++;
			if (p[k] != p[j]) {
				next[j] = k;
			} else {
				next[j] = next[k];
			}
		} else {
			k = next[k];
		}
	}
	return next;
}

int KmpSearch(string s, string p) {
	int *next = GetNext(p);
	int sLen = s.length();
	int pLen = p.length();
	int i = 0;
	int j = 0;
	while (i < sLen && j < pLen) {
		if (j == -1 || s[i] == p[j]) {
			i++;
			j++;
		} else {
			j = next[j];
		}
	}
	if (j == pLen)
		return i - pLen;
	else
		return -1;
}

void solve(string s, string &p, int &pos) {
	int sLen = s.length();
	for (int i = 0; i < sLen / 2; i++) {
		for (int j = sLen / 2 + 1; j >= i + 1; j--) {
			string prep = s.substr(i, j - i);
			string end = s.substr(j, sLen - j);
			if (KmpSearch(end, prep) != -1) {
				p = prep;
				pos = i;
				return;
			}
		}
	}
	pos = -1;
}

int main(int argc, char const *argv[])
{
	string s;
	cin >> s;
	string p;
	int pos;
	solve(s, p, pos);
	if (pos != -1) {
		cout << p << ' ' << pos << '\n';
	} else {
		cout << pos << '\n';
	}
	return 0;
}
```

### 时间复杂度分析
主要算法复杂度为O(n^2), 而其中的kmp算法时间复杂度为O(n),则整个算法复杂度为O(n^3)。
