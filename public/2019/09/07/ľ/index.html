求两个简单链表的并集

问题：

给出两个单向链表，求出它们的并集 例如：

a = [1, 2, 3], b = [2, 3, 4]; 实现InterSect(a, b), 使得a = [1, 2, 3, 4]

代码实现

#include <stdio.h>

typedef struct node {
	int value;
	node *next;
} LinkList;

LinkList* InitList(int n) {
	LinkList *head, *node, *end;
	head = new LinkList;
	end = head;
	while (n--) {
		node = new LinkList;
		scanf("%d", &node->value);
		end->next = node;
		end = node;
	}
	end->next = NULL;
	return head;
}

void ShowList(LinkList *h) {
	while (h->next != NULL) {
		h = h->next;
		printf("%d\n", h->value);
	}
}

void Insert(LinkList *h, int v) {
	LinkList *node = new LinkList;
	while (h->next != NULL)
		h = h->next;
	node->value = v;
	h->next = node;
}

void InterSect(LinkList *a, LinkList *b) {
	while (b->next != NULL) {
		LinkList *A = a;
		b = b->next;
		int flag = 1;
		int tmp = 0;
		while (A->next != NULL) {
			A = A->next;
			if (A->value == b->value) {
				flag = 0;
				break;
			}
		}
		tmp = b->value;
		if (flag)
			Insert(a, tmp);
	}
}

int main() {
	int n;
	scanf("%d", &n);
	LinkList *a = InitList(n);
	LinkList *b = InitList(n);
	ShowList(a);
	ShowList(b);
	InterSect(a, b);
	ShowList(a);
	return 0;
}
